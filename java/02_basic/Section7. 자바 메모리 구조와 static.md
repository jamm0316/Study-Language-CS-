package static2.ex;

        //privateë¡œ ì¸ìŠ¤í„´ìŠ¤ ìƒì„±ì„ ë§‰ëŠ”ë‹¤.

            }
        }
        return minValue;
    }

                maxValue = i;
            }
        }
        return maxValue;
    }
